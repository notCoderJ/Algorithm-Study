        import sys
        import heapq
        input = sys.stdin.readline

        n = int(input())
        pq = []

        for _ in range(n):
            nums = list(map(int, input().split()))
            for num in nums:
                heapq.heappush(pq, num)
            while len(pq) > n:
                heapq.heappop(pq)

        print(heapq.heappop(pq))
